import sys
input = sys.stdin.readline

def count_pairs_with_sum(arr, n, k):
    # Initialize a frequency dictionary to track seen elements
    frequency = {}
    # Initialize result counter
    count = 0
    
    # Iterate through each element in the array
    for i in range(n):
        # Calculate the complement needed to reach target sum
        complement = k - arr[i]
        
        # Check if complement exists in frequency map
        if complement in frequency:
            # Add the frequency of complement to result count
            count += frequency[complement]
        
        # Update the frequency of current element
        if arr[i] in frequency:
            frequency[arr[i]] += 1
        else:
            frequency[arr[i]] = 1
    
    # Return the total count of valid pairs
    return count

def main():
    # Read first line containing N and K
    line = input().split()
    n, k = int(line[0]), int(line[1])
    
    # Read the array elements
    arr = list(map(int, input().split()))
    
    # Calculate result and output
    result = count_pairs_with_sum(arr, n, k)
    print(result)

if __name__ == '__main__':
    main()
